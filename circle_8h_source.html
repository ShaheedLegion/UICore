49  Circlex(Type x, Type y, Type radius) : position(x, y), radius(radius) { }
50  Circlex(const Vec2<Type> &p, Type radius) : position(p), radius(radius) { }
51  Circlex() : position(), radius(static_cast<Type> (0)) { }
52  Circlex(const Circlex<Type> &copy) : position(copy.position), radius(copy.radius) {}
53 
54  bool is_inside(const Vec2<Type> &point) { return radius >= position.distance(point); }
55 
56  Circlex<Type> &operator = (const Circlex<Type>& copy) { position = copy.position; radius = copy.radius; return *this; }
57  bool operator == (const Circlex<Type>& circle) const { return ((position == circle.position) && (radius == circle.radius)); }
58  bool operator != (const Circlex<Type>& circle) const { return ((position != circle.position) || (radius != circle.radius)); }
59  };
