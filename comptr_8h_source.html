37  public:
38  ComPtr() : ptr(0) { }

40  ComPtr(const ComPtr &copy) : ptr(copy.ptr) { if (ptr) ptr->AddRef(); }
41  ~ComPtr() { clear(); }
42  ComPtr &operator =(const ComPtr &copy)
43  {
44  if (this == &copy)
45  return *this;
46  if (copy.ptr)
47  copy.ptr->AddRef();
48  if (ptr)
49  ptr->Release();
50  ptr = copy.ptr;
51  return *this;
52  }
